20 /**
21  * @file rt/src/chthreads.c
22  * @brief Threads code.
23  *
24  * @addtogroup threads
25  * @details Threads related APIs and services.
26  * <h2>Operation mode</h2>
27  * A thread is an abstraction of an independent instructions flow.
28  * In ChibiOS/RT a thread is represented by a "C" function owning
29  * a processor context, state informations and a dedicated stack
30  * area. In this scenario static variables are shared among all
31  * threads while automatic variables are local to the thread.<br>
32  * Operations defined for threads:
33  * - <b>Create</b>, a thread is started on the specified thread
34  * function. This operation is available in multiple variants,
35  * both static and dynamic.
36  * - <b>Exit</b>, a thread terminates by returning from its top
37  * level function or invoking a specific API, the thread can
38  * return a value that can be retrieved by other threads.
39  * - <b>Wait</b>, a thread waits for the termination of another
40  * thread and retrieves its return value.
41  * - <b>Resume</b>, a thread created in suspended state is started.
42  * - <b>Sleep</b>, the execution of a thread is suspended for the
43  * specified amount of time or the specified future absolute time
44  * is reached.
45  * - <b>SetPriority</b>, a thread changes its own priority level.
46  * - <b>Yield</b>, a thread voluntarily renounces to its time slot.
47  * .
48  * @{
49  */

77 /**
78  * @brief Initializes a thread structure.
79  * @note This is an internal functions, do not use it in application code.
80  *
81  * @param[in] oip pointer to the OS instance
82  * @param[in] tp pointer to the thread
83  * @param[in] name thread name
84  * @param[in] prio the priority level for the new thread
85  * @return The same thread pointer passed as parameter.
86  *
87  * @notapi
88  */
89 thread_t *__thd_object_init(os_instance_t *oip,
90  thread_t *tp,
91  const char *name,
92  tprio_t prio) {
93 
94  tp->hdr.pqueue.prio = prio;
95  tp->state = CH_STATE_WTSTART;
96  tp->flags = CH_FLAG_MODE_STATIC;
97  tp->owner = oip;
98 #if CH_CFG_TIME_QUANTUM > 0
99  tp->ticks = (tslices_t)CH_CFG_TIME_QUANTUM;
100 #endif
101 #if CH_CFG_USE_MUTEXES == TRUE
102  tp->realprio = prio;
103  tp->mtxlist = NULL;
104 #endif
105 #if CH_CFG_USE_EVENTS == TRUE
106  tp->epending = (eventmask_t)0;
107 #endif
108 #if CH_DBG_THREADS_PROFILING == TRUE
109  tp->time = (systime_t)0;
110 #endif
111 #if CH_CFG_USE_REGISTRY == TRUE
112  tp->refs = (trefs_t)1;
113  tp->name = name;
114  REG_INSERT(oip, tp);
115 #else
116  (void)name;
117 #endif
118 #if CH_CFG_USE_WAITEXIT == TRUE
119  ch_list_init(&tp->waiting);
120 #endif
121 #if CH_CFG_USE_MESSAGES == TRUE
122  ch_queue_init(&tp->msgqueue);
123 #endif
124 #if CH_DBG_STATISTICS == TRUE
125  chTMObjectInit(&tp->stats);
126 #endif
127  CH_CFG_THREAD_INIT_HOOK(tp);
128  return tp;
129 }

148 /**
149  * @brief Creates a new thread into a static memory area.
150  * @details The new thread is initialized but not inserted in the ready list,
151  * the initial state is @p CH_STATE_WTSTART.
152  * @post The created thread has a reference counter set to one, it is
153  * caller responsibility to call @p chThdRelease() or @p chthdWait()
154  * in order to release the reference. The thread persists in the
155  * registry until its reference counter reaches zero.
156  * @post The initialized thread can be subsequently started by invoking
157  * @p chThdStart(), @p chThdStartI() or @p chSchWakeupS()
158  * depending on the execution context.
159  * @note A thread can terminate by calling @p chThdExit() or by simply
160  * returning from its main function.
161  * @note Threads created using this function do not obey to the
162  * @p CH_DBG_FILL_THREADS debug option because it would keep
163  * the kernel locked for too much time.
164  *
165  * @param[out] tdp pointer to the thread descriptor
166  * @return The pointer to the @p thread_t structure allocated for
167  * the thread into the working space area.
168  *
169  * @iclass
170  */
171 thread_t *chThdCreateSuspendedI(const thread_descriptor_t *tdp) {
172  thread_t *tp;
173 
174  chDbgCheckClassI();
175  chDbgCheck(tdp != NULL);
176  chDbgCheck(MEM_IS_ALIGNED(tdp->wbase, PORT_WORKING_AREA_ALIGN) &&
177  MEM_IS_ALIGNED(tdp->wend, PORT_STACK_ALIGN) &&
178  (tdp->wend > tdp->wbase) &&
179  (((size_t)tdp->wend - (size_t)tdp->wbase) >= THD_WORKING_AREA_SIZE(0)));
180  chDbgCheck((tdp->prio <= HIGHPRIO) && (tdp->funcp != NULL));
181 
182  /* The thread structure is laid out in the upper part of the thread
183  workspace. The thread position structure is aligned to the required
184  stack alignment because it represents the stack top.*/
185  tp = threadref(((uint8_t *)tdp->wend -
186  MEM_ALIGN_NEXT(sizeof (thread_t), PORT_STACK_ALIGN)));
187 
188 #if (CH_DBG_ENABLE_STACK_CHECK == TRUE) || (CH_CFG_USE_DYNAMIC == TRUE)
189  /* Stack boundary.*/
190  tp->wabase = tdp->wbase;
191 #endif
192 
193  /* Setting up the port-dependent part of the working area.*/
194  PORT_SETUP_CONTEXT(tp, tdp->wbase, tp, tdp->funcp, tdp->arg);
195 
196  /* The thread object is initialized but not started.*/
197 #if CH_CFG_SMP_MODE != FALSE
198  if (tdp->instance != NULL) {
199  return __thd_object_init(tdp->instance, tp, tdp->name, tdp->prio);
200  }
201 #endif
202 
203  return __thd_object_init(currcore, tp, tdp->name, tdp->prio);
204 }

309 /**
310  * @brief Creates a new thread into a static memory area.
311  * @post The created thread has a reference counter set to one, it is
312  * caller responsibility to call @p chThdRelease() or @p chthdWait()
313  * in order to release the reference. The thread persists in the
314  * registry until its reference counter reaches zero.
315  * @note A thread can terminate by calling @p chThdExit() or by simply
316  * returning from its main function.
317  *
318  * @param[out] wsp pointer to a working area dedicated to the thread stack
319  * @param[in] size size of the working area
320  * @param[in] prio the priority level for the new thread
321  * @param[in] pf the thread function
322  * @param[in] arg an argument passed to the thread function. It can be
323  * @p NULL.
324  * @return The pointer to the @p thread_t structure allocated for
325  * the thread into the working space area.
326  *
327  * @api
328  */
329 thread_t *chThdCreateStatic(void *wsp, size_t size,
330  tprio_t prio, tfunc_t pf, void *arg) {
331  thread_t *tp;
332 
333  chDbgCheck((wsp != NULL) &&
334  MEM_IS_ALIGNED(wsp, PORT_WORKING_AREA_ALIGN) &&
335  (size >= THD_WORKING_AREA_SIZE(0)) &&
336  MEM_IS_ALIGNED(size, PORT_STACK_ALIGN) &&
337  (prio <= HIGHPRIO) && (pf != NULL));
338 
339 #if (CH_CFG_USE_REGISTRY == TRUE) && \
340  ((CH_DBG_ENABLE_STACK_CHECK == TRUE) || (CH_CFG_USE_DYNAMIC == TRUE))
341  chDbgAssert(chRegFindThreadByWorkingArea(wsp) == NULL,
342  "working area in use");
343 #endif
344 
345 #if CH_DBG_FILL_THREADS == TRUE
346  __thd_stackfill((uint8_t *)wsp, (uint8_t *)wsp + size);
347 #endif
348 
349  chSysLock();
350 
351  /* The thread structure is laid out in the upper part of the thread
352  workspace. The thread position structure is aligned to the required
353  stack alignment because it represents the stack top.*/
354  tp = threadref(((uint8_t *)wsp + size -
355  MEM_ALIGN_NEXT(sizeof (thread_t), PORT_STACK_ALIGN)));
356 
357 #if (CH_DBG_ENABLE_STACK_CHECK == TRUE) || (CH_CFG_USE_DYNAMIC == TRUE)
358  /* Stack boundary.*/
359  tp->wabase = (stkalign_t *)wsp;
360 #endif
361 
362  /* Setting up the port-dependent part of the working area.*/
363  PORT_SETUP_CONTEXT(tp, wsp, tp, pf, arg);
364 
365  tp = __thd_object_init(currcore, tp, "noname", prio);
366 
367  /* Starting the thread immediately.*/
368  chSchWakeupS(tp, MSG_OK);
369  chSysUnlock();
370 
371  return tp;
372 }

415 /**
416  * @brief Releases a reference to a thread object.
417  * @details If the references counter reaches zero <b>and</b> the thread
418  * is in the @p CH_STATE_FINAL state then the thread's memory is
419  * returned to the proper allocator and the thread is removed
420  * from the registry.<br>
421  * Threads whose counter reaches zero and are still active become
422  * "detached" and will be removed from registry on termination.
423  * @pre The configuration option @p CH_CFG_USE_REGISTRY must be enabled in
424  * order to use this function.
425  * @note Static threads are not affected.
426  *
427  * @param[in] tp pointer to the thread
428  *
429  * @api
430  */
431 void chThdRelease(thread_t *tp) {
432 
433  chSysLock();
434  chDbgAssert(tp->refs > (trefs_t)0, "not referenced");
435  tp->refs--;
436 
437  /* If the references counter reaches zero and the thread is in its
438  terminated state then the memory can be returned to the proper
439  allocator.*/
440  if ((tp->refs == (trefs_t)0) && (tp->state == CH_STATE_FINAL)) {
441  REG_REMOVE(tp);
442  chSysUnlock();
443 
444 #if CH_CFG_USE_DYNAMIC == TRUE
445  switch (tp->flags & CH_FLAG_MODE_MASK) {
446 #if CH_CFG_USE_HEAP == TRUE
447  case CH_FLAG_MODE_HEAP:
448  chHeapFree(chThdGetWorkingAreaX(tp));
449  break;
450 #endif
451 #if CH_CFG_USE_MEMPOOLS == TRUE
452  case CH_FLAG_MODE_MPOOL:
453  chPoolFree(tp->mpool, chThdGetWorkingAreaX(tp));
454  break;
455 #endif
456  default:
457  /* Nothing else to do for static threads.*/
458  break;
459  }
460 #endif /* CH_CFG_USE_DYNAMIC == TRUE */
461  return;
462  }
463  chSysUnlock();
464 }
465 #endif /* CH_CFG_USE_REGISTRY == TRUE */

488 /**
489  * @brief Terminates the current thread.
490  * @details The thread goes in the @p CH_STATE_FINAL state holding the
491  * specified exit status code, other threads can retrieve the
492  * exit status code by invoking the function @p chThdWait().
493  * @post Exiting a non-static thread that does not have references
494  * (detached) causes the thread to remain in the registry.
495  * It can only be removed by performing a registry scan operation.
496  * @post Eventual code after this function will never be executed,
497  * this function never returns. The compiler has no way to
498  * know this so do not assume that the compiler would remove
499  * the dead code.
500  *
501  * @param[in] msg thread exit code
502  *
503  * @sclass
504  */
505 void chThdExitS(msg_t msg) {
506  thread_t *currtp = chThdGetSelfX();
507 
508  /* Storing exit message.*/
509  currtp->u.exitcode = msg;
510 
511  /* Exit handler hook.*/
512  CH_CFG_THREAD_EXIT_HOOK(tp);
513 
514 #if CH_CFG_USE_WAITEXIT == TRUE
515  /* Waking up any waiting thread.*/
516  while (unlikely(ch_list_notempty(&currtp->waiting))) {
517  (void) chSchReadyI(threadref(ch_list_unlink(&currtp->waiting)));
518  }
519 #endif
520 
521 #if CH_CFG_USE_REGISTRY == TRUE
522  if (unlikely(currtp->refs == (trefs_t)0)) {
523 #if CH_CFG_USE_DYNAMIC == TRUE
524  /* Static threads are immediately removed from the registry because there
525  is no memory to recover.*/
526  if (unlikely(((currtp->flags & CH_FLAG_MODE_MASK) == CH_FLAG_MODE_STATIC))) {
527  REG_REMOVE(currtp);
528  }
529 #else
530  REG_REMOVE(currtp);
531 #endif
532  }
533 #endif
534 
535  /* Going into final state.*/
536  chSchGoSleepS(CH_STATE_FINAL);
537 
538  /* The thread never returns here.*/
539  chDbgAssert(false, "zombies apocalypse");
540 }

542 #if (CH_CFG_USE_WAITEXIT == TRUE) || defined(__DOXYGEN__)
543 /**
544  * @brief Blocks the execution of the invoking thread until the specified
545  * thread terminates then the exit code is returned.
546  * @details This function waits for the specified thread to terminate then
547  * decrements its reference counter, if the counter reaches zero then
548  * the thread working area is returned to the proper allocator and
549  * the thread is removed from registry.
550  * @pre The configuration option @p CH_CFG_USE_WAITEXIT must be enabled in
551  * order to use this function.
552  * @post Enabling @p chThdWait() requires 2-4 (depending on the
553  * architecture) extra bytes in the @p thread_t structure.
554  * @note If @p CH_CFG_USE_DYNAMIC is not specified this function just waits
555  * for the thread termination, no memory allocators are involved.
556  *
557  * @param[in] tp pointer to the thread
558  * @return The exit code from the terminated thread.
559  *
560  * @api
561  */
562 msg_t chThdWait(thread_t *tp) {
563  thread_t *currtp = chThdGetSelfX();
564  msg_t msg;
565 
566  chDbgCheck(tp != NULL);
567 
568  chSysLock();
569  chDbgAssert(tp != currtp, "waiting self");
570 #if CH_CFG_USE_REGISTRY == TRUE
571  chDbgAssert(tp->refs > (trefs_t)0, "no references");
572 #endif
573 
574  if (likely(tp->state != CH_STATE_FINAL)) {
575  ch_list_link(&tp->waiting, &currtp->hdr.list);
576  chSchGoSleepS(CH_STATE_WTEXIT);
577  }
578  msg = tp->u.exitcode;
579  chSysUnlock();
580 
581 #if CH_CFG_USE_REGISTRY == TRUE
582  /* Releasing a reference to the thread.*/
583  chThdRelease(tp);
584 #endif
585 
586  return msg;
587 }
588 #endif /* CH_CFG_USE_WAITEXIT */

590 /**
591  * @brief Changes the running thread priority level then reschedules if
592  * necessary.
593  * @note The function returns the real thread priority regardless of the
594  * current priority that could be higher than the real priority
595  * because the priority inheritance mechanism.
596  *
597  * @param[in] newprio the new priority level of the running thread
598  * @return The old priority level.
599  *
600  * @api
601  */
602 tprio_t chThdSetPriority(tprio_t newprio) {
603  thread_t *currtp = chThdGetSelfX();
604  tprio_t oldprio;
605 
606  chDbgCheck(newprio <= HIGHPRIO);
607 
608  chSysLock();
609 #if CH_CFG_USE_MUTEXES == TRUE
610  oldprio = currtp->realprio;
611  if ((currtp->hdr.pqueue.prio == currtp->realprio) ||
612  (newprio > currtp->hdr.pqueue.prio)) {
613  currtp->hdr.pqueue.prio = newprio;
614  }
615  currtp->realprio = newprio;
616 #else
617  oldprio = currtp->hdr.pqueue.prio;
618  currtp->hdr.pqueue.prio = newprio;
619 #endif
620  chSchRescheduleS();
621  chSysUnlock();
622 
623  return oldprio;
624 }
