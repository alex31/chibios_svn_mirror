58 /**
59  * @brief Alarm time setup.
60  * @note An RFCU fault is registered if the system time skips past
61  * <tt>(now + delay)</tt>, the deadline is skipped forward
62  * in order to compensate for the event.
63  *
64  * @param[in] now last known system time
65  * @param[in] delay delay over @p now
66  */
67 static void vt_set_alarm(systime_t now, sysinterval_t delay) {
68  sysinterval_t currdelta;
69 
70  /* Initial delta is what is configured statically.*/
71  currdelta = (sysinterval_t)CH_CFG_ST_TIMEDELTA;
72 
73  if (delay < currdelta) {
74  /* We need to avoid that the system time goes past the alarm we are
75  going to set before the alarm is actually set.*/
76  delay = currdelta;
77  }
78 #if CH_CFG_INTERVALS_SIZE > CH_CFG_ST_RESOLUTION
79  else if (delay > VT_MAX_DELAY) {
80  /* The delta could be too large for the physical timer to handle
81  this can happen when: sizeof (systime_t) < sizeof (sysinterval_t).*/
82  delay = VT_MAX_DELAY;
83  }
84 #endif
85 
86  /* Deadline skip detection and correction loop.*/
87  while (true) {
88  sysinterval_t nowdelta;
89  systime_t newnow;
90 
91  /* Setting up the alarm on the next deadline.*/
92  port_timer_set_alarm(chTimeAddX(now, delay));
93 
94  /* Check on current time, we need to detect the error condition where
95  current time skipped past the calculated deadline.
96  Note that the "<" condition is intentional, we want to make sure
97  that the alarm is set before the deadline is reached because the
98  comparison could happen on the transition depending on the timer
99  architecture.*/
100  newnow = chVTGetSystemTimeX();
101  nowdelta = chTimeDiffX(now, newnow);
102  if (likely(nowdelta < delay)) {
103  break;
104  }
105 
106  /* Trying again with a more relaxed minimum delta.*/
107  currdelta += (sysinterval_t)1;
108 
109  /* Current time becomes the new "base" time.*/
110  now = newnow;
111  delay = currdelta;
112  }
113 
114 #if !defined(CH_VT_RFCU_DISABLED)
115  /* Checking if a skip occurred.*/
116  if (currdelta > CH_CFG_ST_TIMEDELTA) {
117  chRFCUCollectFaultsI(CH_RFCU_VT_INSUFFICIENT_DELTA);
118  }
119 #else
120  /* Assertions as fallback.*/
121  chDbgAssert(currdelta <= CH_CFG_ST_TIMEDELTA, "insufficient delta");
122 #endif
123 }

125 /**
126  * @brief Inserts a timer as first element in a delta list.
127  * @note This is the special case when the delta list is initially empty.
128  */
129 static void vt_insert_first(virtual_timers_list_t *vtlp,
130  virtual_timer_t *vtp,
131  systime_t now,
132  sysinterval_t delay) {
133  sysinterval_t currdelta;
134 
135  /* The delta list is empty, the current time becomes the new
136  delta list base time, the timer is inserted.*/
137  vtlp->lasttime = now;
138  ch_dlist_insert_after(&vtlp->dlist, &vtp->dlist, delay);
139 
140  /* Initial delta is what is configured statically.*/
141  currdelta = (sysinterval_t)CH_CFG_ST_TIMEDELTA;
142 
143  /* If the requested delay is lower than the minimum safe delta then it
144  is raised to the minimum safe value.*/
145  if (delay < currdelta) {
146  /* We need to avoid that the system time goes past the alarm we are
147  going to set before the alarm is actually set.*/
148  delay = currdelta;
149  }
150 #if CH_CFG_INTERVALS_SIZE > CH_CFG_ST_RESOLUTION
151  else if (delay > VT_MAX_DELAY) {
152  /* The delta could be too large for the physical timer to handle
153  this can happen when: sizeof (systime_t) < sizeof (sysinterval_t).*/
154  delay = VT_MAX_DELAY;
155  }
156 #endif
157 
158  /* Being the first element inserted in the list the alarm timer
159  is started.*/
160  port_timer_start_alarm(chTimeAddX(vtlp->lasttime, delay));
161 
162  /* Deadline skip detection and correction loop.*/
163  while (true) {
164  systime_t newnow;
165 
166  /* Check on current time, we need to detect the error condition where
167  current time skipped past the calculated deadline.
168  Note that the "<" condition is intentional, we want to make sure
169  that the alarm is set before the deadline is reached because the
170  comparison could happen on the transition depending on the timer
171  architecture.*/
172  newnow = chVTGetSystemTimeX();
173  if (likely(chTimeDiffX(now, newnow) < delay)) {
174  break;
175  }
176 
177  /* Trying again with a more relaxed minimum delta.*/
178  currdelta += (sysinterval_t)1;
179 
180  /* Setting up the alarm on the next deadline.*/
181  port_timer_set_alarm(chTimeAddX(now, currdelta));
182  }
183 
184 #if !defined(CH_VT_RFCU_DISABLED)
185  /* Checking if a skip occurred.*/
186  if (currdelta > CH_CFG_ST_TIMEDELTA) {
187  chRFCUCollectFaultsI(CH_RFCU_VT_INSUFFICIENT_DELTA);
188  }
189 #else
190  /* Assertions as fallback.*/
191  chDbgAssert(currdelta <= CH_CFG_ST_TIMEDELTA, "insufficient delta");
192 #endif
193 }
194 #endif /* CH_CFG_ST_TIMEDELTA > 0 */

196 /**
197  * @brief Enqueues a virtual timer in a virtual timers list.
198  */
199 static void vt_enqueue(virtual_timers_list_t *vtlp,
200  virtual_timer_t *vtp,
201  sysinterval_t delay) {
202  sysinterval_t delta;
203 
204 #if CH_CFG_ST_TIMEDELTA > 0
205  {
206  sysinterval_t nowdelta;
207  systime_t now = chVTGetSystemTimeX();
208 
209  /* Special case where the timers list is empty.*/
210  if (ch_dlist_isempty(&vtlp->dlist)) {
211 
212  vt_insert_first(vtlp, vtp, now, delay);
213 
214  return;
215  }
216 
217  /* Delay as delta from 'lasttime'. Note, it can overflow and the value
218  becomes lower than 'deltanow'.*/
219  nowdelta = chTimeDiffX(vtlp->lasttime, now);
220  delta = nowdelta + delay;
221 
222  /* Scenario where a very large delay exceeded the numeric range, the
223  delta is shortened to make it fit the numeric range, the timer
224  will be triggered "deltanow" cycles earlier.*/
225  if (delta < nowdelta) {
226  delta = delay;
227  }
228 
229  /* Checking if this timer would become the first in the delta list, this
230  requires changing the current alarm setting.*/
231  if (delta < vtlp->dlist.next->delta) {
232 
233  vt_set_alarm(now, delay);
234  }
235  }
236 #else /* CH_CFG_ST_TIMEDELTA == 0 */
237 
238  /* Delta is initially equal to the specified delay.*/
239  delta = delay;
240 #endif /* CH_CFG_ST_TIMEDELTA == 0 */
241 
242  ch_dlist_insert(&vtlp->dlist, &vtp->dlist, delta);
243 }

255  *

257  * @param[in] delay the number of ticks before the operation timeouts, the
258  * special values are handled as follow:
259  * - @a TIME_INFINITE is allowed but interpreted as a
260  * normal time specification.
261  * - @a TIME_IMMEDIATE this value is not allowed.
262  * .

325 /**
326  * @brief Disables a Virtual Timer.
327  * @pre The timer must be in armed state before calling this function.
328  *
329  * @param[in] vtp pointer to a @p virtual_timer_t structure
330  *
331  * @iclass
332  */
333 void chVTDoResetI(virtual_timer_t *vtp) {
334  virtual_timers_list_t *vtlp = &currcore->vtlist;
335 
336  chDbgCheckClassI();
337  chDbgCheck(vtp != NULL);
338  chDbgAssert(chVTIsArmedI(vtp), "timer not armed");
339 
340 #if CH_CFG_ST_TIMEDELTA == 0
341 
342  /* The delta of the timer is added to the next timer.*/
343  vtp->dlist.next->delta += vtp->dlist.delta;
344 
345  /* Removing the element from the delta list, marking it as not armed.*/
346  (void) ch_dlist_dequeue(&vtp->dlist);
347  vtp->dlist.next = NULL;
348 
349  /* The above code changes the value in the header when the removed element
350  is the last of the list, restoring it.*/
351  vtlp->dlist.delta = (sysinterval_t)-1;
352 #else /* CH_CFG_ST_TIMEDELTA > 0 */
353  systime_t now;
354  sysinterval_t nowdelta, delta;
355 
356  /* If the timer is not the first of the list then it is simply unlinked
357  else the operation is more complex.*/
358  if (!ch_dlist_isfirst(&vtlp->dlist, &vtp->dlist)) {
359 
360  /* Removing the element from the delta list.*/
361  (void) ch_dlist_dequeue(&vtp->dlist);
362 
363  /* Adding delta to the next element, if it is not the last one.*/
364  vtp->dlist.next->delta += vtp->dlist.delta;
365 
366  /* Marking timer as not armed.*/
367  vtp->dlist.next = NULL;
368 
369  /* Special case when the removed element from the last position in the list,
370  the value in the header must be restored, just doing it is faster than
371  checking then doing.*/
372  vtlp->dlist.delta = (sysinterval_t)-1;
373 
374  return;
375  }
376 
377  /* Removing the first timer from the list, marking it as not armed.*/
378  ch_dlist_remove_first(&vtlp->dlist);
379  vtp->dlist.next = NULL;
380 
381  /* If the list become empty then the alarm timer is stopped and done.*/
382  if (ch_dlist_isempty(&vtlp->dlist)) {
383 
384  port_timer_stop_alarm();
385 
386  return;
387  }
388 
389  /* The delta of the removed timer is added to the new first timer.*/
390  vtlp->dlist.next->delta += vtp->dlist.delta;
391 
392  /* Distance in ticks between the last alarm event and current time.*/
393  now = chVTGetSystemTimeX();
394  nowdelta = chTimeDiffX(vtlp->lasttime, now);
395 
396  /* If the current time surpassed the time of the next element in list
397  then the event interrupt is already pending, just return.*/
398  if (nowdelta >= vtlp->dlist.next->delta) {
399  return;
400  }
401 
402  /* Distance from the next scheduled event and now.*/
403  delta = vtlp->dlist.next->delta - nowdelta;
404 
405  /* Setting up the alarm.*/
406  vt_set_alarm(now, delta);
407 #endif /* CH_CFG_ST_TIMEDELTA > 0 */
408 }

410 /**
411  * @brief Returns the remaining time interval before next timer trigger.
412  * @note This function can be called while the timer is active.
413  *
414  * @param[in] vtp pointer to a @p virtual_timer_t structure
415  * @return The remaining time interval.
416  *
417  * @iclass
418  */
419 sysinterval_t chVTGetRemainingIntervalI(virtual_timer_t *vtp) {
420  virtual_timers_list_t *vtlp = &currcore->vtlist;
421  sysinterval_t delta;
422  ch_delta_list_t *dlp;
423 
424  chDbgCheckClassI();
425 
426  delta = (sysinterval_t)0;
427  dlp = vtlp->dlist.next;
428  do {
429  delta += dlp->delta;
430  if (dlp == &vtp->dlist) {
431 #if CH_CFG_ST_TIMEDELTA > 0
432  systime_t now = chVTGetSystemTimeX();
433  sysinterval_t nowdelta = chTimeDiffX(vtlp->lasttime, now);
434  if (nowdelta > delta) {
435  return (sysinterval_t)0;
436  }
437  return delta - nowdelta;
438 #else
439  return delta;
440 #endif
441  }
442  dlp = dlp->next;
443  } while (dlp != &vtlp->dlist);
444 
445  chDbgAssert(false, "timer not in list");
446 
447  return (sysinterval_t)-1;
448 }

450 /**
451  * @brief Virtual timers ticker.
452  * @note The system lock is released before entering the callback and
453  * re-acquired immediately after. It is callback's responsibility
454  * to acquire the lock if needed. This is done in order to reduce
455  * interrupts jitter when many timers are in use.
456  *
457  * @iclass
458  */
459 void chVTDoTickI(void) {
460  virtual_timers_list_t *vtlp = &currcore->vtlist;
461 
462  chDbgCheckClassI();
463 
464 #if CH_CFG_ST_TIMEDELTA == 0
465  vtlp->systime++;
466  if (ch_dlist_notempty(&vtlp->dlist)) {
467  /* The list is not empty, processing elements on top.*/
468  --vtlp->dlist.next->delta;
469  while (vtlp->dlist.next->delta == (sysinterval_t)0) {
470  virtual_timer_t *vtp;
471 
472  /* Triggered timer.*/
473  vtp = (virtual_timer_t *)vtlp->dlist.next;
474 
475  /* Removing the element from the delta list, marking it as not armed.*/
476  (void) ch_dlist_dequeue(&vtp->dlist);
477  vtp->dlist.next = NULL;
478 
479  chSysUnlockFromISR();
480  vtp->func(vtp, vtp->par);
481  chSysLockFromISR();
482 
483  /* If a reload is defined the timer needs to be restarted.*/
484  if (vtp->reload > (sysinterval_t)0) {
485  ch_dlist_insert(&vtlp->dlist, &vtp->dlist, vtp->reload);
486  }
487  }
488  }
489 #else /* CH_CFG_ST_TIMEDELTA > 0 */
490  virtual_timer_t *vtp;
491  sysinterval_t nowdelta;
492  systime_t now;
493 
494  /* Looping through timers consuming all timers with deltas lower or equal
495  than the interval between "now" and "lasttime".*/
496  while (true) {
497  systime_t lasttime;
498 
499  /* First timer in the delta list.*/
500  vtp = (virtual_timer_t *)vtlp->dlist.next;
501 
502  /* Delta between current time and last execution time.*/
503  now = chVTGetSystemTimeX();
504  nowdelta = chTimeDiffX(vtlp->lasttime, now);
505 
506  /* Loop break condition.
507  Note that the list scan is limited by the delta list header having
508  "vtlp->dlist.delta == (sysinterval_t)-1" which is greater than all
509  deltas*/
510  if (nowdelta < vtp->dlist.delta) {
511  break;
512  }
513 
514  /* Last time deadline is updated to the next timer's time.*/
515  lasttime = chTimeAddX(vtlp->lasttime, vtp->dlist.delta);
516  vtlp->lasttime = lasttime;
517 
518  /* Removing the timer from the list, marking it as not armed.*/
519  (void) ch_dlist_dequeue(&vtp->dlist);
520  vtp->dlist.next = NULL;
521 
522  /* If the list becomes empty then the alarm is disabled.*/
523  if (ch_dlist_isempty(&vtlp->dlist)) {
524  port_timer_stop_alarm();
525  }
526 
527  /* The callback is invoked outside the kernel critical section, it
528  is re-entered on the callback return. Note that "lasttime" can be
529  modified within the callback if some timer function is called.*/
530  chSysUnlockFromISR();
531 
532  vtp->func(vtp, vtp->par);
533 
534  chSysLockFromISR();
535 
536  /* If a reload is defined the timer needs to be restarted.*/
537  if (unlikely(vtp->reload > (sysinterval_t)0)) {
538  sysinterval_t delta, delay;
539 
540  /* Refreshing the now delta after spending time in the callback for
541  a more accurate detection of too fast reloads.*/
542  now = chVTGetSystemTimeX();
543  nowdelta = chTimeDiffX(lasttime, now);
544 
545 #if !defined(CH_VT_RFCU_DISABLED)
546  /* Checking if the required reload is feasible.*/
547  if (nowdelta > vtp->reload) {
548  /* System time is already past the deadline, logging the fault and
549  proceeding with a minimum delay.*/
550 
551  chDbgAssert(false, "skipped deadline");
552  chRFCUCollectFaultsI(CH_RFCU_VT_SKIPPED_DEADLINE);
553 
554  delay = (sysinterval_t)0;
555  }
556  else {
557  /* Enqueuing the timer again using the calculated delta.*/
558  delay = vtp->reload - nowdelta;
559  }
560 #else
561  /* Assertions as fallback.*/
562  chDbgAssert(nowdelta <= vtp->reload, "skipped deadline");
563 
564  /* Enqueuing the timer again using the calculated delta.*/
565  delay = vtp->reload - nowdelta;
566 #endif
567 
568  /* Special case where the timers list is empty.*/
569  if (ch_dlist_isempty(&vtlp->dlist)) {
570 
571  vt_insert_first(vtlp, vtp, now, delay);
572 
573  return;
574  }
575 
576  /* Delay as delta from 'lasttime'. Note, it can overflow and the value
577  becomes lower than 'nowdelta'. In that case the delta is shortened
578  to make it fit the numeric range and the timer will be triggered
579  "nowdelta" cycles earlier.*/
580  delta = nowdelta + delay;
581  if (delta < nowdelta) {
582  delta = delay;
583  }
584 
585  /* Insert into delta list. */
586  ch_dlist_insert(&vtlp->dlist, &vtp->dlist, delta);
587  }
588  }
589 
590  /* If the list is empty, nothing else to do.*/
591  if (ch_dlist_isempty(&vtlp->dlist)) {
592  return;
593  }
594 
595  /* The "unprocessed nowdelta" time slice is added to "last time"
596  and subtracted to next timer's delta.*/
597  vtlp->lasttime += nowdelta;
598  vtp->dlist.delta -= nowdelta;
599 
600  /* Update alarm time to next timer.*/
601  vt_set_alarm(now, vtp->dlist.delta);
602 #endif /* CH_CFG_ST_TIMEDELTA > 0 */
603 }

605 #if (CH_CFG_USE_TIMESTAMP == TRUE) || defined(__DOXYGEN__)
606 /**
607  * @brief Generates a monotonic time stamp.
608  * @details This function generates a monotonic time stamp synchronized with
609  * the system time. The time stamp has the same resolution of
610  * system time.
611  * @note There is an assumption, this function must be called at
612  * least once before the system time wraps back to zero or
613  * synchronization is lost. You may use a periodic virtual timer with
614  * a very large interval in order to keep time stamps synchronized
615  * by calling this function.
616  *
617  * @return The time stamp.
618  *
619  * @iclass
620  */
621 systimestamp_t chVTGetTimeStampI(void) {
622  os_instance_t * oip = currcore;
623  systimestamp_t last, stamp;
624  systime_t now;
625 
626  chDbgCheckClassI();
627 
628  /* Current system time.*/
629  now = chVTGetSystemTimeX();
630 
631  /* Last time stamp generated.*/
632  last = oip->vtlist.laststamp;
633 
634  /* Interval between the last time stamp and current time used for a new
635  time stamp. Note that this fails if the interval is larger than a
636  systime_t type.*/
637  stamp = last + (systimestamp_t)chTimeDiffX((systime_t)last, now);
638 
639  chDbgAssert(oip->vtlist.laststamp <= stamp, "wrapped");
640 
641  /* Storing the new stamp.*/
642  oip->vtlist.laststamp = stamp;
643 
644  return stamp;
645 }
