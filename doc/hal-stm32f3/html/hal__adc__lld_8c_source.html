108 /**
109  * @brief Enables the ADC voltage regulator.
110  *
111  * @param[in] adcp pointer to the @p ADCDriver object
112  */
113 static void adc_lld_vreg_on(ADCDriver *adcp) {
114 
115 #if defined(STM32F3XX)
116  adcp->adcm->CR = 0; /* RM 12.4.3.*/
117  adcp->adcm->CR = ADC_CR_ADVREGEN_0;
118 #if STM32_ADC_DUAL_MODE
119  adcp->adcs->CR = ADC_CR_ADVREGEN_0;
120 #endif
121  osalSysPolledDelayX(OSAL_US2RTC(STM32_HCLK, 10));
122 #endif
123 
124 #if defined(STM32L4XX)
125  adcp->adcm->CR = 0; /* RM 16.3.6.*/
126  adcp->adcm->CR = ADC_CR_ADVREGEN;
127 #if STM32_ADC_DUAL_MODE
128  adcp->adcs->CR = ADC_CR_ADVREGEN;
129 #endif
130  osalSysPolledDelayX(OSAL_US2RTC(STM32_HCLK, 20));
131 #endif
132 }

160 /**
161  * @brief Enables the ADC analog circuit.
162  *
163  * @param[in] adcp pointer to the @p ADCDriver object
164  */
165 static void adc_lld_analog_on(ADCDriver *adcp) {
166 
167 #if defined(STM32F3XX)
168  adcp->adcm->CR |= ADC_CR_ADEN;
169  while ((adcp->adcm->ISR & ADC_ISR_ADRD) == 0)
170  ;
171 #if STM32_ADC_DUAL_MODE
172  adcp->adcs->CR |= ADC_CR_ADEN;
173  while ((adcp->adcs->ISR & ADC_ISR_ADRD) == 0)
174  ;
175 #endif
176 #endif
177 
178 #if defined(STM32L4XX)
179  adcp->adcm->CR |= ADC_CR_ADEN;
180  while ((adcp->adcm->ISR & ADC_ISR_ADRDY) == 0)
181  ;
182 #if STM32_ADC_DUAL_MODE
183  adcp->adcs->CR |= ADC_CR_ADEN;
184  while ((adcp->adcs->ISR & ADC_ISR_ADRDY) == 0)
185  ;
186 #endif
187 #endif
188 }

241 /**
242  * @brief Stops an ongoing conversion, if any.
243  *
244  * @param[in] adcp pointer to the @p ADCDriver object
245  */
246 static void adc_lld_stop_adc(ADCDriver *adcp) {
247 
248  if (adcp->adcm->CR & ADC_CR_ADSTART) {
249  adcp->adcm->CR |= ADC_CR_ADSTP;
250  while (adcp->adcm->CR & ADC_CR_ADSTP)
251  ;

255 /**
256  * @brief ADC DMA ISR service routine.
257  *
258  * @param[in] adcp pointer to the @p ADCDriver object
259  * @param[in] flags pre-shifted content of the ISR register
260  */
261 static void adc_lld_serve_dma_interrupt(ADCDriver *adcp, uint32_t flags) {
262 
263  /* DMA errors handling.*/
264  if ((flags & (STM32_DMA_ISR_TEIF | STM32_DMA_ISR_DMEIF)) != 0) {
265  /* DMA, this could help only if the DMA tries to access an unmapped
266  address space or violates alignment rules.*/
267  _adc_isr_error_code(adcp, ADC_ERR_DMAFAILURE);
268  }
269  else {
270  /* It is possible that the conversion group has already be reset by the
271  ADC error handler, in this case this interrupt is spurious.*/
272  if (adcp->grpp != NULL) {
273  if ((flags & STM32_DMA_ISR_TCIF) != 0) {
274  /* Transfer complete processing.*/
275  _adc_isr_full_code(adcp);
276  }
277  else if ((flags & STM32_DMA_ISR_HTIF) != 0) {
278  /* Half transfer processing.*/
279  _adc_isr_half_code(adcp);
280  }
281  }
282  }
283 }

285 /**
286  * @brief ADC ISR service routine.
287  *
288  * @param[in] adcp pointer to the @p ADCDriver object
289  * @param[in] isr content of the ISR register
290  */
291 static void adc_lld_serve_interrupt(ADCDriver *adcp, uint32_t isr) {
292 
293  /* It could be a spurious interrupt caused by overflows after DMA disabling,
294  just ignore it in this case.*/
295  if (adcp->grpp != NULL) {
296  /* Note, an overflow may occur after the conversion ended before the driver
297  is able to stop the ADC, this is why the DMA channel is checked too.*/
298  if ((isr & ADC_ISR_OVR) &&
299  (dmaStreamGetTransactionSize(adcp->dmastp) > 0)) {
300  /* ADC overflow condition, this could happen only if the DMA is unable
301  to read data fast enough.*/
302  _adc_isr_error_code(adcp, ADC_ERR_OVERFLOW);
303  }
304  if (isr & ADC_ISR_AWD1) {
305  /* Analog watchdog error.*/
306  _adc_isr_error_code(adcp, ADC_ERR_AWD1);
307  }
308  if (isr & ADC_ISR_AWD2) {
309  /* Analog watchdog error.*/
310  _adc_isr_error_code(adcp, ADC_ERR_AWD2);
311  }
312  if (isr & ADC_ISR_AWD3) {
313  /* Analog watchdog error.*/
314  _adc_isr_error_code(adcp, ADC_ERR_AWD3);
315  }
316  }
317 }

574 /**
575  * @brief Configures and activates the ADC peripheral.
576  *
577  * @param[in] adcp pointer to the @p ADCDriver object
578  *
579  * @notapi
580  */
581 void adc_lld_start(ADCDriver *adcp) {
582 
583  /* Handling the default configuration.*/
584  if (adcp->config == NULL) {
585  adcp->config = &default_config;
586  }
587 
588  /* If in stopped state then enables the ADC and DMA clocks.*/
589  if (adcp->state == ADC_STOP) {
590 #if STM32_ADC_USE_ADC1
591  if (&ADCD1 == adcp) {
592  bool b;
593  b = dmaStreamAllocate(adcp->dmastp,
594  STM32_ADC_ADC1_DMA_IRQ_PRIORITY,
595  (stm32_dmaisr_t)adc_lld_serve_dma_interrupt,
596  (void *)adcp);
597  osalDbgAssert(!b, "stream already allocated");
598 
599  clkmask |= (1 << 0);
600 #if defined(STM32F3XX)
601  rccEnableADC12(true);
602 #endif
603 #if defined(STM32L4XX)
604  rccEnableADC123(true);
605 #endif
606  }
607 #endif /* STM32_ADC_USE_ADC1 */
608 
609 #if STM32_ADC_USE_ADC2
610  if (&ADCD2 == adcp) {
611  bool b;
612  b = dmaStreamAllocate(adcp->dmastp,
613  STM32_ADC_ADC2_DMA_IRQ_PRIORITY,
614  (stm32_dmaisr_t)adc_lld_serve_dma_interrupt,
615  (void *)adcp);
616  osalDbgAssert(!b, "stream already allocated");
617 
618  clkmask |= (1 << 1);
619 #if defined(STM32F3XX)
620  rccEnableADC12(true);
621 #endif
622 #if defined(STM32L4XX)
623  rccEnableADC123(true);
624 #endif
625  }
626 #endif /* STM32_ADC_USE_ADC2 */

779 /**
780  * @brief Starts an ADC conversion.
781  *
782  * @param[in] adcp pointer to the @p ADCDriver object
783  *
784  * @notapi
785  */
786 void adc_lld_start_conversion(ADCDriver *adcp) {
787  uint32_t dmamode, cfgr;
788  const ADCConversionGroup *grpp = adcp->grpp;
789 #if STM32_ADC_DUAL_MODE
790  uint32_t ccr = grpp->ccr & ~(ADC_CCR_CKMODE_MASK | ADC_CCR_MDMA_MASK);
791 #endif
792 
793  osalDbgAssert(!STM32_ADC_DUAL_MODE || ((grpp->num_channels & 1) == 0),
794  "odd number of channels in dual mode");
795 
796  /* Calculating control registers values.*/
797  dmamode = adcp->dmamode;
798  cfgr = grpp->cfgr | ADC_CFGR_DMAEN;
799  if (grpp->circular) {
800  dmamode |= STM32_DMA_CR_CIRC;
801 #if STM32_ADC_DUAL_MODE
802  ccr |= ADC_CCR_DMACFG_CIRCULAR;
803 #else
804  cfgr |= ADC_CFGR_DMACFG_CIRCULAR;
805 #endif
806  if (adcp->depth > 1) {
807  /* If circular buffer depth > 1, then the half transfer interrupt
808  is enabled in order to allow streaming processing.*/
809  dmamode |= STM32_DMA_CR_HTIE;
810  }
811  }
812 
813  /* DMA setup.*/
814  dmaStreamSetMemory0(adcp->dmastp, adcp->samples);
815 #if STM32_ADC_DUAL_MODE
816  dmaStreamSetTransactionSize(adcp->dmastp, ((uint32_t)grpp->num_channels/2) *
817  (uint32_t)adcp->depth);
818 #else
819  dmaStreamSetTransactionSize(adcp->dmastp, (uint32_t)grpp->num_channels *
820  (uint32_t)adcp->depth);
821 #endif
822  dmaStreamSetMode(adcp->dmastp, dmamode);
823  dmaStreamEnable(adcp->dmastp);
824 
825  /* ADC setup, if it is defined a callback for the analog watch dog then it
826  is enabled.*/
827  adcp->adcm->ISR = adcp->adcm->ISR;
828  adcp->adcm->IER = ADC_IER_OVR | ADC_IER_AWD1;
829  adcp->adcm->TR1 = grpp->tr1;
830 #if STM32_ADC_DUAL_MODE
831 
832  /* Configuring the CCR register with the user-specified settings
833  in the conversion group configuration structure, static settings are
834  preserved.*/
835  adcp->adcc->CCR = (adcp->adcc->CCR &
836  (ADC_CCR_CKMODE_MASK | ADC_CCR_MDMA_MASK)) | ccr;
837 
838  adcp->adcm->SMPR1 = grpp->smpr[0];
839  adcp->adcm->SMPR2 = grpp->smpr[1];
840  adcp->adcm->SQR1 = grpp->sqr[0] | ADC_SQR1_NUM_CH(grpp->num_channels / 2);
841  adcp->adcm->SQR2 = grpp->sqr[1];
842  adcp->adcm->SQR3 = grpp->sqr[2];
843  adcp->adcm->SQR4 = grpp->sqr[3];
844  adcp->adcs->SMPR1 = grpp->ssmpr[0];
845  adcp->adcs->SMPR2 = grpp->ssmpr[1];
846  adcp->adcs->SQR1 = grpp->ssqr[0] | ADC_SQR1_NUM_CH(grpp->num_channels / 2);
847  adcp->adcs->SQR2 = grpp->ssqr[1];
848  adcp->adcs->SQR3 = grpp->ssqr[2];
849  adcp->adcs->SQR4 = grpp->ssqr[3];
850 
851 #else /* !STM32_ADC_DUAL_MODE */
852  adcp->adcm->SMPR1 = grpp->smpr[0];
853  adcp->adcm->SMPR2 = grpp->smpr[1];
854  adcp->adcm->SQR1 = grpp->sqr[0] | ADC_SQR1_NUM_CH(grpp->num_channels);
855  adcp->adcm->SQR2 = grpp->sqr[1];
856  adcp->adcm->SQR3 = grpp->sqr[2];
857  adcp->adcm->SQR4 = grpp->sqr[3];
858 #endif /* !STM32_ADC_DUAL_MODE */
859 
860  /* ADC configuration.*/
861  adcp->adcm->CFGR = cfgr;
862 #if (STM32_ADCV3_OVERSAMPLING == TRUE) || defined(__DOXYGEN__)
863  adcp->adcm->CFGR2 = grpp->cfgr2;
864 #endif
865 
866  /* Starting conversion.*/
867  adcp->adcm->CR |= ADC_CR_ADSTART;
868 }
